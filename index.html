const moveZeros = (arr) => {
  const zeros = [];
  for (let x of arr) {
    if (x === 0) {
      zeros.push(x);
    }
  };
  const leftArr = arr.filter(x => x !== 0);

  return [...leftArr, ...zeros];
};
